drop table level;
CREATE TABLE level (
	level int PRIMARY KEY, 
	parcours char(40), 
	sceduled datetime,
	gametime int
);  

drop table game;
CREATE TABLE game (
	id int not null  auto_increment,
	level int references level, 
	game int, 
	bot1 char(20) references bot, 
	bot2 char(20) references bot, 
	sceduled datetime, 
	finishtime int, 
	state char(20) default 'not init', 
	winner char(20) references bot, 
	screenshot char(20),
	primary key (id)
);

drop table bot;
CREATE TABLE bot (
	bot char(20) PRIMARY KEY, 
	team char(20) not null references team, 
	bin char(40)
);

drop table team;
CREATE TABLE team (
	team char(20) PRIMARY KEY, 
	email char(30)
);

drop table log;
CREATE TABLE log (
	id int not null  auto_increment,
	game int references game,
	logtime int,
	pos1x float,
	pos1y float,
	head1x float,
	head1y float,
	state1 int,
	pos2x float,
	pos2y float,
	head2x float,
	head2y float, 
	state2 int,
	primary key (id)
);
